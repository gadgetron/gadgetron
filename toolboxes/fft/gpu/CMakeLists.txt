  ADD_DEFINITIONS(-D__BUILD_GADGETRON_GPUFFT__)

  ${CMAKE_SOURCE_DIR}/toolboxes/core/gpu
  ${CUDA_INCLUDE_DIRS}
  ${Boost_INCLUDE_DIR}
)
